# 查看远程分支
git remote

# 删除指定的远程仓库
git remote remove <remote-name>

# 或者使用 rm 命令（效果相同）
git remote rm <remote-name>

branches ​

  • git checkout main #切换到 main 分支
  • git branch #查看本地分支
  • git branch -a #查看所有分支（包括远程）
  • git branch -r #仅查看远程分支
