    class Solution {
    public int removeElement(int[] nums, int val) {
        Arrays.sort(nums);
        int flagIndex = -1;
        int count = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] == val && count == 0) {
                flagIndex = i;
                count++;
            } else if (nums[i] == val) {
                count++;
            }
        }
        if (flagIndex != -1 && count != 0) {
            for (int i = flagIndex; i < flagIndex + count; i++) {
                nums[i] = nums[nums.length - (i - flagIndex) - 1];
            }
        }
        return nums.length - count;
    }
}

    class Solution {

    public int climbStairs(int n) {
        if (n == 0)
            return 0;
        if (n == 1)
            return 1;
        if (n == 2)
            return 2;
        int[] dp = new int[n + 1];
        dp[0] = 0;
        dp[1] = 1;
        dp[2] = 2;
        for (int i = 3; i <= n; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        return dp[n];
    }

}

    public class Solution80 {
    public static int removeDuplicates(int[] nums) {
        if (nums.length <= 2) return nums.length;
        int slow = -1;
        int fast = 1;
        int count = 1;
        while (slow <= fast && fast < nums.length) {
            if (nums[fast] == nums[fast - 1]) {
                count++;
                if (count > 2 && slow == -1) {
                    slow = fast;
                } else if (count <= 2 && slow != -1) {
                    nums[slow] = nums[fast];
                    slow++;
                }
            } else if (nums[fast] != nums[fast - 1]) {
                if (slow != -1) {
                    nums[slow] = nums[fast];
                    slow++;
                }
                count = 1;
            }
            fast++;
        }
        return slow == -1 ? nums.length : slow;
    }

    public static void main(String[] args) {
        //int[] nums = {0,0,1,1,1,1,2,3,3};
        //int[] nums = {1,2,2};
        int[] nums = {1, 1, 1, 2, 2, 2, 3, 3};
        for (int num : nums) {
            System.out.print(num + " ");
        }
        System.out.println();
        System.out.println(removeDuplicates(nums));
        for (int num : nums) {
            System.out.print(num + " ");
        }
    }
}

    public class Solution61 {

    public static ListNode rotateRight(ListNode head, int k) {
        if (head == null || head.next == null)  return head;
        ListNode _head = head;
        int nodeCount = 1;
        while(_head.next != null) {
            _head = _head.next;
            nodeCount++;
        }
        _head.next = head;
        _head = head;
        for (int i = 0; i < (nodeCount - k % nodeCount) - 1; i++) {
            _head = _head.next;
        }
        head = _head.next;
        _head.next = null;
        return head;
    }

    public static void main(String[] args) {
        test002();
        test001();
    }

    public static void test001() {
        ListNode node0 = new ListNode(1);
        ListNode node1 = new ListNode(2);
        ListNode node2 = new ListNode(3);
        ListNode node3 = new ListNode(4);
        ListNode node4 = new ListNode(5);
        node0.next = node1;
        node1.next = node2;
        node2.next = node3;
        node3.next = node4;
        ListNode listNode = rotateRight(node0, 2);
        while(listNode != null) {
            System.out.print(listNode.val + " ");
            listNode = listNode.next;
        }
        System.out.println();
    }

    public static void test002() {
        ListNode node0 = new ListNode(0);
        ListNode node1 = new ListNode(1);

        node0.next = node1;

            System.out.print(listNode.val + " ");

        }
        System.out.println();
    }

    public class Solution86 {

    public static ListNode partition(ListNode head, int x) {
        ListNode solder = new ListNode(-1);
        solder.next = head;
        ListNode insertPreNode = solder;
        ListNode insertNode = insertPreNode.next;
        while (insertNode != null && insertNode.val < x) {
            insertPreNode = insertNode;
            insertNode = insertNode.next;
        }
        ListNode preNode = insertPreNode;
        ListNode currentNode = preNode.next;
        ListNode nextNode = currentNode != null ? currentNode.next : null;
        while (currentNode != null) {
            if (currentNode.val < x) {
                preNode.next = nextNode;
                insertPreNode.next = currentNode;
                currentNode.next = insertNode;
                insertPreNode = currentNode;
            } else {
                preNode = currentNode;
            }
            currentNode = nextNode;
            nextNode = currentNode != null ? currentNode.next : null;
        }
        return solder.next;
    }

    public static void main(String[] args) {
        test002();
        test001();
    }

    public static void test001() {

        ListNode node0 = new ListNode(1);
        ListNode node1 = new ListNode(4);
        ListNode node2 = new ListNode(3);
        ListNode node3 = new ListNode(2);
        ListNode node4 = new ListNode(5);
        ListNode node5 = new ListNode(2);
        node0.next = node1;
        node1.next = node2;
        node2.next = node3;
        node3.next = node4;
        node4.next = node5;
        ListNode listNode = partition(node0, 3);
        while(listNode != null) {
            System.out.print(listNode.val + " ");
            listNode = listNode.next;
        }
        System.out.println();

    }

            System.out.print(listNode.val + " ");
            listNode = listNode.next;
        }
        System.out.println();

    }

        }

        ListNode(int val) {
            this.val = val;
        }

        ListNode(int val, ListNode next) {
            this.val = val;
