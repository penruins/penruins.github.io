i = 3;
f(i) = f(i - nums[0]) + f(i - nums[1]) + f(i - nums[2]);
f(0) = 1;
f(1) = f(0) + f(-1) + f(-2) = 1;
f(2) = f(1) + f(0) + f(-1) = 2;
f(3) = f(2) + f(1) + f(0) = 4;
